def calculate_ppl(model, data):
    ppls = []
    for item in data:
        _,ppl = model(item)
        ppls.append(ppl)
    return ppls

def calculate_accuracy(threshold,ppl,label):
    predicted=np.array([1 if x>threshold else 0 for x in ppl])
    actual = np.array(label)
    accuracy = np.mean(predicted == actual)
    return accuracy
